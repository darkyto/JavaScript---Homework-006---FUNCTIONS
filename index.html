Problem 001 - English digit

Write a function that returns the last digit of given integer as an English word.

function isInt(n){
    return Number(n)===n && n%1===0;
}
var lastToWord = function (num) {
	if (!isNaN(num)) {
		var result = '';
		if (isInt(num)) {
			num = Math.abs(num); // for the negative values..
			var last = num % 10;
			var words = ['zero','one','two','three','four','five','six','seven','eight','nine'];
			for (var i = 0; i < words.length; i++) {
				result = words[last];
			}
			return result;
		}
		else {
			return 'Float numbers not allowed!';
		}
	}
	else {
		return 'Wrong input - only integers allowed!';
	}
};

Problem 002 - Reverse number

Write a function that reverses the digits of given decimal number.

	function reverseNum(num) {
		var str = num.toString();
		var result = str.split('').reverse().join('');
		result = result * 1;
		return result;
	}

Problem 003 - Occurrences of word

Write a function that finds all the occurrences of word in a text. The search can be case sensitive or case insensitive. Use function overloading.

	function getMatches (text, word, caseSense) {
		if (caseSense || caseSense === undefined) {
			word = word.toLowerCase();
			text = text.toLowerCase();
		}
		var ind = 0;
		var wordLength = word.length;
		var index = [];
		var matches = [];

		while ( (index = text.indexOf(word, ind)) > -1 ) {
			matches.push(index);
			ind = index + wordLength;
		}
		return matches;
	}

	var matchNumber = function (arr, num) {

	var compareWithNeigh = function(arr, i) {
		if (i === 0 || i === arr.length-1) {
			return false + ' element do not have two neigh.';

		else {
